"# checkforteamupdates" 

# Run Docker conatiner on AWS
# REF: https://us-east-2.console.aws.amazon.com/ecs/home?region=us-east-2#/firstRun

# PRE-REQ:
# Install / update AWS CLI - https://docs.aws.amazon.com/cli/latest/userguide/installing.html
pip install awscli --upgrade --user


# Part II - Build, tag, and push Docker image
#############################################

# 1) Retrieve the docker login command that you can use to authenticate your Docker client to your registry: 
aws ecr get-login --no-include-email --region us-east-2

# 2) Run the docker login command that was returned in the previous step

# 3) Build your Docker image
docker build -t checkforteamupdates2018 .

# 4) After the build completes, tag your image so you can push the image to this repository
docker tag checkforteamupdates2018:latest BLAH.dkr.ecr.us-east-2.amazonaws.com/checkforteamupdates2018:latest

# 5) Run the following command to push this image to your newly created AWS repository
docker push BLAH.dkr.ecr.us-east-2.amazonaws.com/checkforteamupdates2018:latest
